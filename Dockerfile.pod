FROM nvidia/cuda:12.1.1-cudnn8-devel-ubuntu22.04

ENV DEBIAN_FRONTEND=noninteractive \
    PIP_NO_CACHE_DIR=1 \
    PIP_BREAK_SYSTEM_PACKAGES=1 \
    PYTHONUNBUFFERED=1 \
    PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True

RUN apt-get update && apt-get install -y --no-install-recommends \
    software-properties-common \
    && add-apt-repository ppa:deadsnakes/ppa \
    && apt-get update && apt-get install -y --no-install-recommends \
    python3.12 python3.12-venv python3.12-dev \
    git ca-certificates curl build-essential ffmpeg libsndfile1 \
    && rm -rf /var/lib/apt/lists/* \
    && curl -sS https://bootstrap.pypa.io/get-pip.py | python3.12 \
    && ln -sf /usr/bin/python3.12 /usr/local/bin/python \
    && ln -sf /usr/bin/python3.12 /usr/local/bin/python3

WORKDIR /workspace/vibevoice

RUN pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu121

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt
RUN pip install "runpod>=1.6.0"
RUN pip install "fastapi>=0.110" "uvicorn[standard]>=0.27" "websockets>=12"

RUN pip install git+https://github.com/vibevoice-community/VibeVoice.git

COPY startup.py handler.py inference.py config.py bootstrap.sh pod_server.py ./

EXPOSE 8000

# Create a pod startup script that creates voice file and runs pod_server
RUN echo '#!/usr/bin/env python3' > /workspace/vibevoice/pod_entrypoint.py && \
    echo 'import os, sys, wave' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'sys.path.insert(0, "/workspace/vibevoice")' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'import config' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'os.makedirs(config.AUDIO_PROMPTS_DIR, exist_ok=True)' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'vp = os.path.join(config.AUDIO_PROMPTS_DIR, "Alice.wav")' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'if not os.path.exists(vp):' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo '    with wave.open(vp, "wb") as f:' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo '        f.setnchannels(1); f.setsampwidth(2); f.setframerate(24000)' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo '        f.writeframes(bytes(144000))' >> /workspace/vibevoice/pod_entrypoint.py && \
    echo 'os.execvp("python3", ["python3", "/workspace/vibevoice/pod_server.py"])' >> /workspace/vibevoice/pod_entrypoint.py

CMD ["python3", "/workspace/vibevoice/pod_entrypoint.py"]
